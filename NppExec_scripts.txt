for Python:

NPP_SAVE
cmd /k echo. & Python -u "$(FULL_CURRENT_PATH)" & ECHO. & PAUSE & EXIT

for c++:(project)

NPP_SAVE
cd $(CURRENT_DIRECTORY)
cmd /k qmake *.pro &  mingw32-make & echo off & cd release &  for %x in (*.exe) do ( start "" /b  %x )  &  echo. &  echo.  & ECHO. & pause >nul & EXIT 
NPP_CONSOLE 0


for c++:(single file)

NPP_CONSOLE 0
NPP_SAVE 
npp_run cmd /c   " cd $(CURRENT_DIRECTORY) & del $(NAME_PART).exe >nul 2>&1 & g++ -Wall -std=c++11 $(NAME_PART).cpp -o  $(NAME_PART) & echo. & $(NAME_PART).exe  &  echo. &  echo. & echo. & pause"
